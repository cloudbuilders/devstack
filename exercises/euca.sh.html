euca.sh

we will use the euca2ools cli tool that wraps the python boto library to test ec2 compatibility

This script exits on an error so that errors don't compound and you see only the first error that occured.

#!/usr/bin/env bash

Print the commands being run so that we can see the command that triggers an error. It is also useful for following allowing as the install occurs.

set -o errexit

Settings

set -o xtrace

Use openrc + stackrc + localrc for settings

Find a machine image to boot

pushd $(cd $(dirname "$0")/.. && pwd)
source ./openrc
popd

Define secgroup

IMAGE=`euca-describe-images | grep machine | cut -f2 | head -n1`

Add a secgroup

SECGROUP=euca_secgroup

Launch it

euca-add-group -d description $SECGROUP

Assure it has booted within a reasonable time

INSTANCE=`euca-run-instances -g $SECGROUP -t m1.tiny $IMAGE | grep INSTANCE | cut -f2`

Allocate floating address

if ! timeout $RUNNING_TIMEOUT sh -c "while euca-describe-instances $INSTANCE | grep -q running; do sleep 1; done"; then
    echo "server didn't become active within $RUNNING_TIMEOUT seconds"
    exit 1
fi

Release floating address

FLOATING_IP=`euca-allocate-address | cut -f2`

Authorize pinging

euca-associate-address -i $INSTANCE $FLOATING_IP

Max time till the vm is bootable

euca-authorize -P icmp -s 0.0.0.0/0 -t -1:-1 $SECGROUP

Revoke pinging

BOOT_TIMEOUT=${BOOT_TIMEOUT:-15}
if ! timeout $BOOT_TIMEOUT sh -c "while ! ping -c1 -w1 $FLOATING_IP; do sleep 1; done"; then
    echo "Couldn't ping server"
    exit 1
fi

Delete group

euca-revoke -P icmp -s 0.0.0.0/0 -t -1:-1 $SECGROUP

Release floating address

euca-delete-group $SECGROUP

Release floating address

euca-disassociate-address $FLOATING_IP

Terminate instance

euca-release-address $FLOATING_IP


euca-terminate-instances $INSTANCE
